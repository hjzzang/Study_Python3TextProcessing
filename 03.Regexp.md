```python
# 정규표현식
import re
p = re.compile('[a.b]')

m = p.match('zab')
print(m)

s = p.search('zaccccb')
print(s)

s = p.search('a?c')
print(s)
s.span()

p = re.compile('[a-z]+')
m = p.match("3 python")
print(m.group())

s = p.search("3 python is good")
print(s.group())
p.findall("3 python is good")
print(p.finditer("3 python is good"))
```
